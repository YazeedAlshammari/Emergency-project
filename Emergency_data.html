Q3:Do certain seasons correlate with an increase or decrease in patient admissions?

In [24]:
# Create a new DataFrame with seasons as columns
seasons_mapping = {
    'Winter': ['December', 'January', 'February'],
    'Spring': ['March', 'April', 'May'],
    'Summer': ['June', 'July', 'August'],
    'Fall': ['September', 'October', 'November']
}

# Create a new DataFrame with seasons as columns
seasons_df = pd.DataFrame()
